### 项目结构
```
src/
├── main/
│   ├── database/
│   │   └── index.ts      # SQLite 模块
│   └── index.ts          # 主进程入口
├── preload/
│   ├── index.ts          # Preload 脚本
│   └── index.d.ts        # 类型定义
└── renderer/
    └── src/
        ├── store/
        │   ├── index.ts  # Redux store
        │   └── hooks.ts  # Typed hooks
        ├── routes/
        │   └── index.tsx # React Router
        └── main.tsx      # 渲染进程入口
```

### 使用方式

**Redux:**
```tsx
import { useAppSelector, useAppDispatch } from './store/hooks'
```

**React Router:**
```tsx
import { useNavigate, useParams } from 'react-router-dom'
```

**TailwindCSS:**
```tsx
<div className="flex items-center justify-center bg-blue-500">
```

**SQLite (渲染进程):**
```tsx
// 执行 SQL
await window.api.db.execute('CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY)')

// 查询多条
const users = await window.api.db.query<User>('SELECT * FROM users')

// 查询单条
const user = await window.api.db.queryOne<User>('SELECT * FROM users WHERE id = ?', [1])
```
